# A codeblock with defaults

```{#id0 .stitch}
#! /usr/bin/env bash
echo -e "# stderr\r\r$0" >&2
set -x
echo -e "# stdout\r\r$0" >&1
echo -e "# artifact\r\r$0" > $1
```

# read as gfm markdown

```{#id1 .stitch inc=err!gfm}
#! /usr/bin/env bash
echo -e "# stderr\r\n$0" >&2
set -x
echo -e "# stdout\r\n$0" >&1
echo -e "# artifact\r\n$0" > $1
```
